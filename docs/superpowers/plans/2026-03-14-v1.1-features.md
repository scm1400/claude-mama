# v1.1 Features Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add Always on Top toggle, contextual (situation-aware) messages, achievement badges, and custom character skins to Claude Mama.

**Architecture:** Four independent features layered onto the existing Electron + React app. Each feature follows the same pattern: types → core logic (pure, testable) → main process integration → renderer UI. Features B and C both extend `TriggerContext` with `resetsAt`, so B must land before C.

**Tech Stack:** Electron 40, React 19, TypeScript 5.9, Vite 7, Vitest, electron-store

**Spec:** `docs/superpowers/specs/2026-03-14-v1.1-features-design.md`

```typescript
import { TriggerContext, ContextTrigger, MamaMood, Locale, DailyUtilRecord } from '../shared/types';
import { getMessage } from './messages';

/**
 * Evaluates context triggers in priority order.
 * Returns the highest-priority matching trigger, or null.
 */
export function evaluateContextTrigger(ctx: TriggerContext): ContextTrigger | null {
  const { now, dailyHistory, weeklyUtilization, resetsAt } = ctx;

  // Priority 1: unusedStreak (N >= 2 consecutive zero-usage days)
  if (dailyHistory.length >= 2) {
    const sorted = [...dailyHistory].sort((a, b) => b.date.localeCompare(a.date));
    let streak = 0;
    for (const d of sorted) {
      if (d.percent === 0) streak++;
      else break;
    }
    if (streak >= 2) return 'unusedStreak';
  }

  // Priority 2: resetImminent (reset in <3 hours + weekly <50%)
  if (resetsAt && weeklyUtilization !== null && weeklyUtilization < 50) {
    const resetTime = new Date(resetsAt).getTime();
    const nowTime = now.getTime();
    const hoursLeft = (resetTime - nowTime) / (1000 * 60 * 60);
    if (hoursLeft > 0 && hoursLeft <= 3) return 'resetImminent';
  }

  // Priority 3: spike (30%+ increase from yesterday)
  if (dailyHistory.length >= 2) {
    const sorted = [...dailyHistory].sort((a, b) => b.date.localeCompare(a.date));
    if (sorted[0] && sorted[1] && sorted[0].percent - sorted[1].percent >= 30) {
      return 'spike';
    }
  }

  // Priority 4: weekend
  const day = now.getDay();
  if (day === 0 || day === 6) return 'weekend';

  return null;
}
```

```typescript
import { BadgeTriggerContext } from '../shared/types';

export function evaluateBadgeTriggers(ctx: BadgeTriggerContext): string[] {
  const triggered: string[] = [];
  const { weeklyUtilization, dailyHistory, installDate, firstApiCallSeen, now, proudCount, angryCount } = ctx;

  // Bronze: first call
  if (!firstApiCallSeen && weeklyUtilization !== null && weeklyUtilization > 0) {
    triggered.push('badge_first_call');
  }

  // Streak calculation
  if (dailyHistory.length > 0) {
    const sorted = [...dailyHistory].sort((a, b) => b.date.localeCompare(a.date));
    let streak = 0;
    for (const d of sorted) {
      if (d.percent > 0) streak++;
      else break;
    }
    if (streak >= 3) triggered.push('badge_streak_3');
    if (streak >= 7) triggered.push('badge_streak_7');
    if (streak >= 30) triggered.push('badge_streak_30');
  }

  // Days since install
  const installDt = new Date(installDate);
  const daysSince = Math.floor((now.getTime() - installDt.getTime()) / (24 * 60 * 60 * 1000));
  if (daysSince >= 7) triggered.push('badge_mama_7days');

  // Usage milestones
  if (weeklyUtilization !== null && weeklyUtilization >= 50) triggered.push('badge_half');
  if (weeklyUtilization !== null && weeklyUtilization >= 100) triggered.push('badge_full');

  // Mood counts
  if (proudCount >= 10) triggered.push('badge_proud_10');
  if (angryCount >= 10 && weeklyUtilization !== null && weeklyUtilization > 0) {
    triggered.push('badge_survivor');
  }

  return triggered;
}
```

```typescript
import { UnlockedBadge, BadgeState, BadgeTier } from '../shared/types';
import { BADGE_REGISTRY, getBadgeById } from './badge-registry';

export class BadgeManager {
  private unlocked: Map<string, UnlockedBadge>;

  constructor(persisted: UnlockedBadge[]) {
    this.unlocked = new Map(persisted.map((b) => [b.id, b]));
  }

  unlock(id: string, now: Date): boolean {
    if (this.unlocked.has(id)) return false;
    if (!getBadgeById(id)) return false;
    this.unlocked.set(id, { id, unlockedAt: now.toISOString() });
    return true;
  }

  processTriggered(triggeredIds: string[], now: Date): string[] {
    const newlyUnlocked: string[] = [];
    for (const id of triggeredIds) {
      if (this.unlock(id, now)) {
        newlyUnlocked.push(id);
      }
    }
    return newlyUnlocked;
  }

  getState(): BadgeState {
    const byTier: Record<BadgeTier, { unlocked: number; total: number }> = {
      bronze: { unlocked: 0, total: 0 },
      silver: { unlocked: 0, total: 0 },
      gold: { unlocked: 0, total: 0 },
    };
    for (const entry of BADGE_REGISTRY) {
      byTier[entry.tier].total++;
      if (this.unlocked.has(entry.id)) {
        byTier[entry.tier].unlocked++;
      }
    }
    return {
      unlocked: [...this.unlocked.values()],
      totalCount: BADGE_REGISTRY.length,
      byTier,
    };
  }

  serialize(): UnlockedBadge[] {
    return [...this.unlocked.values()];
  }
}
```

### Task 14: Create skin manager

**Files:**
- Create: `src/main/skin-manager.ts`

- [ ] **Step 1: Implement skin-manager.ts**

```typescript
import { app, dialog } from 'electron';
import fs from 'fs';
import path from 'path';
import { SkinConfig, SkinMode } from '../shared/types';
import { getStore } from './ipc-handlers';

const SKINS_DIR = path.join(app.getPath('userData'), 'skins');
const MAX_FILE_SIZE = 2 * 1024 * 1024; // 2MB
const ALLOWED_EXTENSIONS = ['.png', '.jpg', '.jpeg', '.gif'];

function ensureSkinsDir(): void {
  if (!fs.existsSync(SKINS_DIR)) {
    fs.mkdirSync(SKINS_DIR, { recursive: true });
  }
}

function cleanupOldSkins(keepPaths: string[]): void {
  ensureSkinsDir();
  const files = fs.readdirSync(SKINS_DIR);
  for (const file of files) {
    const fullPath = path.join(SKINS_DIR, file);
    if (!keepPaths.includes(fullPath)) {
      try { fs.unlinkSync(fullPath); } catch { /* ignore */ }
    }
  }
}

export async function uploadSkinImage(mood?: string): Promise<string | null> {
  const result = await dialog.showOpenDialog({
    title: 'Select Character Image',
    filters: [{ name: 'Images', extensions: ['png', 'jpg', 'jpeg', 'gif'] }],
    properties: ['openFile'],
  });

  if (result.canceled || result.filePaths.length === 0) return null;

  const srcPath = result.filePaths[0];
  const ext = path.extname(srcPath).toLowerCase();

  if (!ALLOWED_EXTENSIONS.includes(ext)) return null;

  const stat = fs.statSync(srcPath);
  if (stat.size > MAX_FILE_SIZE) return null;

  ensureSkinsDir();
  const fileName = mood
    ? `skin-${mood}-${Date.now()}${ext}`
    : `skin-${Date.now()}${ext}`;
  const destPath = path.join(SKINS_DIR, fileName);
  fs.copyFileSync(srcPath, destPath);

  return destPath;
}

export function resetSkin(): void {
  const store = getStore();
  (store as any).set('skin', { mode: 'default' });
  cleanupOldSkins([]);
}

export function getSkinConfig(): SkinConfig {
  const store = getStore();
  return (store as any).get('skin', { mode: 'default' }) as SkinConfig;
}

export function saveSkinConfig(config: SkinConfig): void {
  const store = getStore();

  // Collect paths to keep
  const keepPaths: string[] = [];
  if (config.singleImagePath) keepPaths.push(config.singleImagePath);
  if (config.moodImages) {
    Object.values(config.moodImages).forEach((p) => { if (p) keepPaths.push(p); });
  }
  if (config.spritesheet?.imagePath) keepPaths.push(config.spritesheet.imagePath);

  cleanupOldSkins(keepPaths);
  (store as any).set('skin', config);
}
```
